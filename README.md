# HOW TO USE ? 

# IN TREMUX

pkg update

pkg upgrade

pkg install git

pkg install python

git clone https://github.com/giwan-Sec/Mean-Raport

cd Mean-Raport

python main.py OR python3 main.py


# IN KALI LINUX, DEBIAN, OR ETC.

apt-get install python (recommended python3)

git clone https://github.com/giwan-Sec/Mean-Raport

cd Mean-Raport

python main.py OR python3 main.py
